Future<List<T>> fromJsonAsListAsync<T>(List<dynamic> jsonList,
    Future<T> Function(Map<String, dynamic>) fromJson) async {
  List<T> data = [];
  for (final Map<String, dynamic> json in jsonList) {
    data.add(await fromJson(json));
  }
  return data;
}
